#include <stdio.h>

void check_road(int N, int adj_matrix[][N], int A, int B) {
    if (A < 0 || A >= N || B < 0 || B >= N) {
        printf("Invalid city numbers\n");
        return;
    }
    if (adj_matrix[A][B] == 1)
        printf("Connected\n");
    else
        printf("Not connected\n");
}

int main() {
    int N, A, B;
    printf("Enter number of cities N: ");
    scanf("%d", &N);
    int adj_matrix[N][N];
    printf("Enter adjacency matrix (%dx%d):\n", N, N);
    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++)
            scanf("%d", &adj_matrix[i][j]);
    printf("Enter two city numbers A B: ");
    scanf("%d %d", &A, &B);
    check_road(N, adj_matrix, A, B);
    return 0;
}
